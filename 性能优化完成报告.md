## 💻 前端修改要点

### 1. 并行请求图表和AI分析

```javascript
async function startAnalysis() {
    // 并行执行
    const [chartResult, analyzeResult] = await Promise.all([
        fetch('/api/chart-data', {...}),
        fetch('/api/analyze', {...})
    ]);

    // 立即显示图表
    renderChart(chartResult.data);

    // 监听AI进度
    const taskId = analyzeResult.task_id;
    listenToAIProgress(taskId);
}
```

### 2. SSE监听AI进度

```javascript
function listenToAIProgress(taskId) {
    const eventSource = new EventSource(`/api/task/${taskId}/progress`);

    eventSource.onmessage = (event) => {
        const data = JSON.parse(event.data);

        if (data.type === 'done') {
            console.log('✅ 完成');
            eventSource.close();
            return;
        }

        // 更新UI
        updateAgentStatus(data.agent);
        appendReport(data.text);
    };
}
```
